\section{Conclusions}

\begin{frame}[t]{Summary}
\begin{itemize}
  \item A \textgood{unified programming model} for 
        \textmark{sequential} and \textmark{parallel} modes.
  \vfill
  \item Multiple \textgood{back-ends} available:
        \textmark{sequential}, \textmark{OpenMP},
        \textmark{TBB}, \textmark{native}, \textmark{FastFlow}.
  \vfill
  \item Current pattern set:
    \begin{itemize}
      \item \textenum{Data}: \textmark{map}, \textmark{reduce}, \textmark{map/reduce}, \textmark{stencil}.
      \item \textenum{Task}: \textmark{divide/conquer}.
      \item \textenum{Streaming}: \textmark{pipeline} with nesting of \textmark{farm}, \textmark{filter},
            \textmark{reduction}, \textmark{iteration}.
    \end{itemize}

  \vfill
  \item \textgood{Low} programming effort.
  \vfill
  \item \textgood{Very low} performance overhead.
\end{itemize}
\end{frame}

\begin{frame}[t]{Future work}
\begin{itemize}
  \item Integrate additional backends:
  \item Eliminate metaprogramming by using Concepts (C++20).
  \item Extend and simplify the interface for data patterns.
  \item Support multi-context patterns.
  \item Better support of NUMA for native back-end. 
  \item More patterns.
  \item More applications.
\end{itemize}
\end{frame}
